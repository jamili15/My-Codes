import React, {
  useRef,
  ChangeEvent,
  useState,
  InputHTMLAttributes,
  ChangeEventHandler,
  FocusEventHandler,
  useEffect,
} from "react";
import { useKeyboardContext } from "../keyboard/KeyboardContext";

interface InputProps extends InputHTMLAttributes<HTMLInputElement> {
  label?: string;
  className?: string;
  placeholder?: string;
  labelLayout?: string;
  translationUp?: string;
  translationDown?: string;
  type?:
    | "text"
    | "number"
    | "email"
    | "password"
    | "date"
    | "tel"
    | "time"
    | "file"
    | "checkbox"
    | "search";
  onChange?: ChangeEventHandler<HTMLInputElement>;
}

const Input: React.FC<InputProps> = ({
  label,
  className,
  labelLayout,
  translationUp,
  translationDown,
  placeholder,
  type = "text",
  value,
  onChange,
  ...props
}) => {
  const context = useKeyboardContext();
  const inputRef = useRef<HTMLInputElement>(null);
  const [focused, setFocused] = useState(false);
  const [hasValue, setHasValue] = useState(!!value);

  function focusHandler() {
    if (inputRef.current) {
      context.setActiveInput(inputRef);
    }
  }

  const handleFocus: FocusEventHandler<HTMLInputElement> = () => {
    setFocused(true);
  };

  const handleBlur: FocusEventHandler<HTMLInputElement> = () => {
    setFocused(false);
  };

  const handleChange = (event: ChangeEvent<HTMLInputElement>) => {
    const { value } = event.target;
    setHasValue(!!value);

    if (onChange) {
      onChange(event); // Pass the entire event object
    }
  };

  useEffect(() => {
    // Update hasValue when the value prop changes
    setHasValue(!!value);
  }, [value]);

  return (
    <div className="relative">
      <label
        htmlFor={type}
        className={`absolute left-2 px-1 transition-transform ${labelLayout} ${
          (focused || hasValue) && !props.readOnly
            ? `text-blue-500 -translate-y-3 ${translationUp} bg-white rounded !m-0`
            : `text-gray-500 translate-y-2 ${
                hasValue ? "-translate-y-3" : ""
              } ${translationDown}`
        }`}
      >
        {label}
      </label>
      <input
        type={type}
        id={type}
        placeholder={placeholder}
        className={`w-full bg-transparent px-4 py-2 focus:outline-none rounded ${
          focused ? "focus:border-blue-500" : "focus:border-primary"
        } ${className}`}
        ref={inputRef}
        onFocus={(e) => {
          focusHandler();
          handleFocus(e);
        }}
        onBlur={(e) => {
          handleBlur(e);
        }}
        onClick={(e) => e.stopPropagation()}
        onChange={handleChange}
        value={value}
        {...props}
      />
    </div>
  );
};

export default Input;
